# MUFramework Code Generation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现两套代码生成机制——① Source Generator 生成 IOpenArgs 动态分发；② Editor Code Generator（UIBindingCollector + 命名规则 + 代码写出）生成组件自动绑定。可选：UniTask 扩展层。

**Architecture:** SG 为标准 .NET Roslyn Analyzer 项目，检测 `partial class : IOpenArgs<T>` 生成 `.Generated.cs`。Editor Code Generator 为 Unity Editor 脚本，扫描预制体子节点命名规则，序列化引用后写出 `.AutoBind.cs`。

**Tech Stack:** Roslyn Source Generator（.NET 6 SDK）、Unity Editor scripting、C# partial class、可选 UniTask

**先决条件:** Plan 1-3 已完成（IOpenArgs 接口、UIWindow.AutoBindComponents 钩子、UIPanel.AutoBindComponents 钩子均已就位）

```csharp
// Analyzer~/UIOpenArgsGenerator.cs
using System.Collections.Generic;
using System.Linq;
using System.Text;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp;
using Microsoft.CodeAnalysis.CSharp.Syntax;

namespace MUFramework.SourceGenerator
{
    [Generator]
    public class UIOpenArgsGenerator : IIncrementalGenerator
    {
        public void Initialize(IncrementalGeneratorInitializationContext context)
        {
            var candidates = context.SyntaxProvider
                .CreateSyntaxProvider(
                    predicate: (node, _) => node is ClassDeclarationSyntax c
                        && c.Modifiers.Any(SyntaxKind.PartialKeyword)
                        && c.BaseList != null,
                    transform: (ctx, _) => GetClassInfo(ctx))
                .Where(x => x != null);

            context.RegisterSourceOutput(candidates, (spc, info) =>
            {
                if (info == null) return;
                spc.AddSource($"{info.ClassName}.Generated.cs", GenerateSource(info));
            });
        }

        private ClassInfo? GetClassInfo(GeneratorSyntaxContext ctx)
        {
            var classDecl = (ClassDeclarationSyntax)ctx.Node;
            var symbol = ctx.SemanticModel.GetDeclaredSymbol(classDecl) as INamedTypeSymbol;
            if (symbol == null) return null;

            // 收集所有 IOpenArgs<T...> 接口
            var openArgsInterfaces = symbol.AllInterfaces
                .Where(i => i.Name == "IOpenArgs" && i.ContainingNamespace.Name == "MUFramework")
                .ToList();

            if (!openArgsInterfaces.Any()) return null;

            return new ClassInfo
            {
                ClassName = symbol.Name,
                Namespace = symbol.ContainingNamespace.IsGlobalNamespace
                    ? null
                    : symbol.ContainingNamespace.ToDisplayString(),
                Interfaces = openArgsInterfaces.Select(i => new OpenArgsInfo
                {
                    TypeArgs = i.TypeArguments.Select(t => t.ToDisplayString()).ToList()
                }).ToList()
            };
        }

        private string GenerateSource(ClassInfo info)
        {
            var sb = new StringBuilder();
            sb.AppendLine("// <auto-generated/> DO NOT EDIT");
            sb.AppendLine("using System;");
            sb.AppendLine();
            if (info.Namespace != null)
            {
                sb.AppendLine($"namespace {info.Namespace}");
                sb.AppendLine("{");
            }
            sb.AppendLine($"    public partial class {info.ClassName}");
            sb.AppendLine("    {");
            sb.AppendLine("        internal override void OnOpenInternal(object[] args)");
            sb.AppendLine("        {");

            foreach (var iface in info.Interfaces)
            {
                var types = iface.TypeArgs;
                var conditions = new List<string>();
                for (int i = 0; i < types.Count; i++)
                    conditions.Add($"args[{i}] is {types[i]} _a{i}");

                var argList = string.Join(", ", Enumerable.Range(0, types.Count).Select(i => $"_a{i}"));
                var ifaceType = $"IOpenArgs<{string.Join(", ", types)}>";

                sb.AppendLine($"            if (args != null && args.Length == {types.Count} && {string.Join(" && ", conditions)})");
                sb.AppendLine("            {");
                sb.AppendLine($"                (({ifaceType})this).OnOpen({argList}); return;");
                sb.AppendLine("            }");
            }

            sb.AppendLine("            base.OnOpenInternal(args);");
            sb.AppendLine("        }");
            sb.AppendLine("    }");

            if (info.Namespace != null) sb.AppendLine("}");
            return sb.ToString();
        }

        private class ClassInfo
        {
            public string ClassName { get; set; } = "";
            public string? Namespace { get; set; }
            public List<OpenArgsInfo> Interfaces { get; set; } = new();
        }

        private class OpenArgsInfo
        {
            public List<string> TypeArgs { get; set; } = new();
        }
    }
}
```

```csharp
// Assets/MUFramework/Editor/Binding/UIBindingNamingRule.cs
using System;
using System.Collections.Generic;

namespace MUFramework.Editor
{
    /// <summary>
    /// 命名规则抽象基类。继承此类并覆写 PrefixMap 可自定义前缀映射。
    /// 注册到 UIGlobal.BindingNamingRule 后框架自动使用。
    /// </summary>
    public abstract class UIBindingNamingRule
    {
        /// <summary> 节点名前缀 → 组件类型映射 </summary>
        public abstract IReadOnlyDictionary<string, Type> PrefixMap { get; }

        /// <summary>
        /// 节点名 → C# 字段名。默认：Btn_Close → _btnClose
        /// 子类可覆写实现自定义转换。
        /// </summary>
        public virtual string ToFieldName(string nodeName, string matchedPrefix)
        {
            // 去掉前缀，剩余部分首字母小写，整体加下划线前缀
            var suffix = nodeName.Substring(matchedPrefix.Length);
            if (string.IsNullOrEmpty(suffix)) return "_" + matchedPrefix.TrimEnd('_').ToLower();
            return "_" + char.ToLower(matchedPrefix.TrimEnd('_')[0])
                       + matchedPrefix.TrimEnd('_').Substring(1)
                       + suffix;
        }

        /// <summary> 尝试匹配节点名，返回 (fieldName, componentType) 或 null </summary>
        public (string fieldName, Type componentType)? TryMatch(string nodeName)
        {
            foreach (var (prefix, type) in PrefixMap)
            {
                if (nodeName.StartsWith(prefix, StringComparison.OrdinalIgnoreCase))
                    return (ToFieldName(nodeName, prefix), type);
            }
            return null;
        }
    }
}
```

```csharp
// Assets/MUFramework/Editor/Binding/UIBindingCodeGenerator.cs
using System.IO;
using System.Text;
using UnityEditor;
using UnityEngine;

namespace MUFramework.Editor
{
    public static class UIBindingCodeGenerator
    {
        /// <summary>
        /// 根据 UIBindingCollector 数据生成 TargetClassName.AutoBind.cs。
        /// outputDir：生成文件所在目录（Assets 相对路径）。
        /// </summary>
        public static void Generate(UIBindingCollector collector, string outputDir)
        {
            if (collector == null || string.IsNullOrEmpty(collector.TargetClassName)) return;
            if (collector.Entries == null || collector.Entries.Count == 0) return;

            var sb = new StringBuilder();
            sb.AppendLine("// <auto-generated/> DO NOT EDIT — generated by UIBindingCodeGenerator");
            sb.AppendLine("using UnityEngine;");
            sb.AppendLine();

            bool hasNs = !string.IsNullOrEmpty(collector.TargetNamespace);
            if (hasNs) { sb.AppendLine($"namespace {collector.TargetNamespace}"); sb.AppendLine("{"); }

            string indent = hasNs ? "    " : "";
            sb.AppendLine($"{indent}public partial class {collector.TargetClassName}");
            sb.AppendLine($"{indent}{{");

            // 字段声明
            foreach (var entry in collector.Entries)
                sb.AppendLine($"{indent}    private {entry.ComponentType} {entry.FieldName};");

            sb.AppendLine();
            sb.AppendLine($"{indent}    internal override void AutoBindComponents()");
            sb.AppendLine($"{indent}    {{");
            foreach (var entry in collector.Entries)
            {
                if (entry.ComponentType == "UnityEngine.GameObject")
                    sb.AppendLine($"{indent}        {entry.FieldName} = transform.Find(\"{entry.Path}\")?.gameObject;");
                else
                    sb.AppendLine($"{indent}        {entry.FieldName} = transform.Find(\"{entry.Path}\")?.GetComponent<{entry.ComponentType}>();");
            }
            sb.AppendLine($"{indent}    }}");
            sb.AppendLine($"{indent}}}");

            if (hasNs) sb.AppendLine("}");

            // 写文件
            Directory.CreateDirectory(Path.Combine(Application.dataPath, "..", outputDir));
            var filePath = Path.Combine(outputDir, $"{collector.TargetClassName}.AutoBind.cs");
            File.WriteAllText(filePath, sb.ToString(), Encoding.UTF8);
            AssetDatabase.Refresh();
            Debug.Log($"[MUI] Generated: {filePath}");
        }
    }
}
```

- [ ] **Step 1: 创建 UIBindingCollectorEditor.cs**

```csharp
// Assets/MUFramework/Editor/Binding/UIBindingCollectorEditor.cs
using System;
using UnityEditor;
using UnityEngine;

namespace MUFramework.Editor
{
    [CustomEditor(typeof(UIBindingCollector))]
    public class UIBindingCollectorEditor : UnityEditor.Editor
    {
        public override void OnInspectorGUI()
        {
            DrawDefaultInspector();
            var collector = (UIBindingCollector)target;

            EditorGUILayout.Space();
            if (GUILayout.Button("Scan & Generate Bindings"))
            {
                ScanAndGenerate(collector);
            }
        }

        private void ScanAndGenerate(UIBindingCollector collector)
        {
            var rule = UIGlobal.BindingNamingRule;
            if (rule == null)
            {
                Debug.LogError("[MUI] UIGlobal.BindingNamingRule is null.");
                return;
            }

            collector.Entries.Clear();

            // 递归扫描所有子节点
            ScanChildren(collector.transform, collector.transform, collector, rule);

            EditorUtility.SetDirty(collector);

            // 推导输出目录：与 UIBindingCollector 所在 prefab 同目录
            var prefabPath = AssetDatabase.GetAssetPath(
                PrefabUtility.GetCorrespondingObjectFromSource(collector.gameObject)
                ?? collector.gameObject);
            var outputDir = string.IsNullOrEmpty(prefabPath)
                ? "Assets"
                : System.IO.Path.GetDirectoryName(prefabPath);

            UIBindingCodeGenerator.Generate(collector, outputDir);
        }

        private void ScanChildren(Transform root, Transform current,
            UIBindingCollector collector, UIBindingNamingRule rule)
        {
            foreach (Transform child in current)
            {
                var match = rule.TryMatch(child.name);
                if (match.HasValue)
                {
                    var (fieldName, componentType) = match.Value;
                    Component compRef = componentType == typeof(GameObject)
                        ? null
                        : child.GetComponent(componentType);

                    // 计算相对路径
                    var path = GetRelativePath(root, child);

                    collector.Entries.Add(new UIBindingEntry
                    {
                        FieldName = fieldName,
                        ComponentType = componentType.FullName,
                        ComponentRef = compRef,
                        Path = path,
                    });
                }
                // 继续递归子节点
                ScanChildren(root, child, collector, rule);
            }
        }

        private string GetRelativePath(Transform root, Transform target)
        {
            var parts = new System.Collections.Generic.List<string>();
            var current = target;
            while (current != root)
            {
                parts.Insert(0, current.name);
                current = current.parent;
            }
            return string.Join("/", parts);
        }
    }
}
```

```csharp
// Assets/MUFramework/Extensions/UniTask/UIManagerUniTaskExtensions.cs
#if UNITASK_SUPPORT
using System.Threading;
using Cysharp.Threading.Tasks;

namespace MUFramework
{
    public static class UIManagerUniTaskExtensions
    {
        public static UniTask<TWindow> OpenAsync<TWindow>(
            this UIManager manager, CancellationToken ct = default)
            where TWindow : UIWindow
        {
            var tcs = new UniTaskCompletionSource<TWindow>();
            ct.Register(() => tcs.TrySetCanceled());
            manager.OpenAsync<TWindow>(w => tcs.TrySetResult(w));
            return tcs.Task;
        }

        public static UniTask<TWindow> OpenAsync<TWindow, TData>(
            this UIManager manager, TData data, CancellationToken ct = default)
            where TWindow : UIWindow<TData>
        {
            var tcs = new UniTaskCompletionSource<TWindow>();
            ct.Register(() => tcs.TrySetCanceled());
            manager.OpenAsync<TWindow, TData>(data, w => tcs.TrySetResult(w));
            return tcs.Task;
        }

        public static UniTask<TWindow> OpenAsync<TWindow, T1>(
            this UIManager manager, T1 arg1, CancellationToken ct = default)
            where TWindow : UIWindow, IOpenArgs<T1>
        {
            var tcs = new UniTaskCompletionSource<TWindow>();
            ct.Register(() => tcs.TrySetCanceled());
            manager.OpenAsync<TWindow, T1>(arg1, w => tcs.TrySetResult(w));
            return tcs.Task;
        }

        public static UniTask CloseAsync(
            this UIManager manager, long uniqueId, CancellationToken ct = default)
        {
            var tcs = new UniTaskCompletionSource();
            manager.Close(uniqueId, withAnimation: true, onComplete: () => tcs.TrySetResult());
            return tcs.Task;
        }
    }
}
#endif
```
